func main() {
    console.print("Enter an integer: ");
    integer = console.get();
    
    if(valid(integer)) {
        result = calculate(integer);
        console.print("You gave me: " + integer);
        console.print("I got: " + result);
        console.print("All done!");
        program.exit();
    } else {
        console.print("Error: not a valid integer!");
        program.quit();
    }
}

func valid(i) {
    return i.valid();
}

func calculate(x) {
    return math(x);
}
